ALTER PROCEDURE [dbo].[SP_UPD_EQUIPOS]
	@PI_ID_EQUIPO INTEGER,
	@PV_NOMBRE_EQUIPO VARCHAR (100),
	@PV_DESCRIPCION VARCHAR (100),
	@PV_UBICACION VARCHAR (100),
	@PV_OBJETIVO VARCHAR (100),
	@PV_CONTACTO VARCHAR (100),
	@PV_ESTADO VARCHAR (1)
	
AS
BEGIN
	
	UPDATE EQUIPOS
	SET NOMBRE_EQUIPO =@PV_NOMBRE_EQUIPO,
	DESCRIPCION =@PV_DESCRIPCION,
	UBICACION =@PV_UBICACION,
	OBJETIVO=@PV_OBJETIVO,
	CONTACTO=@PV_CONTACTO,
	ESTADO=@PV_ESTADO
	WHERE ID_EQUIPO=@PI_ID_EQUIPO

END
