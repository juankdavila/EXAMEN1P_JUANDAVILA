ALTER PROCEDURE [dbo].[SP_INS_FUTBOLISTA]
	@PI_ID_FUTBOLISTA INTEGER,
	@PV_NOMBRE VARCHAR (100),
	@PV_APELLIDOS VARCHAR (100),
	@PV_EDAD VARCHAR (100),
	@PV_DIRECCION VARCHAR (100),
	@PV_MAIL VARCHAR (100),
	@PD_FECHA_NACIMIENTO DATETIME,
	@PV_ESTADO VARCHAR(1)

	
AS
BEGIN
	
	INSERT INTO FUTBOLISTA(ID_FUTBOLISTA,NOMBRE,APELLIDOS,EDAD,DIRECCION,MAIL,FECHA_NACIMIENTO,ESTADO)
	VALUES(@PI_ID_FUTBOLISTA,@PV_NOMBRE,@PV_APELLIDOS,@PV_EDAD,@PV_DIRECCION,@PV_MAIL,@PD_FECHA_NACIMIENTO,'A');

END
