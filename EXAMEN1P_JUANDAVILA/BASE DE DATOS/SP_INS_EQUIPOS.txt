ALTER PROCEDURE [dbo].[SP_INS_EQUIPOS]
	@PI_ID_EQUIPO INTEGER,
	@PV_NOMBRE_EQUIPO VARCHAR (100),
	@PV_DESCRIPCION VARCHAR (100),
	@PV_UBICACION VARCHAR (100),
	@PV_OBJETIVO VARCHAR (100),
	@PV_CONTACTO VARCHAR (100),
	@PV_ESTADO VARCHAR (1)
	

	
AS
BEGIN
	
	INSERT INTO EQUIPOS(ID_EQUIPO,NOMBRE_EQUIPO,DESCRIPCION,UBICACION,OBJETIVO,CONTACTO,ESTADO)
	VALUES(@PI_ID_EQUIPO,@PV_NOMBRE_EQUIPO,@PV_DESCRIPCION,@PV_UBICACION,@PV_OBJETIVO,@PV_CONTACTO,'A');

END
