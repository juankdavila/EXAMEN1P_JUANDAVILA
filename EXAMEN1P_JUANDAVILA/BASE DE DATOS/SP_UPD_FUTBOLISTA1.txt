ALTER PROCEDURE [dbo].[SP_UPD_FUTBOLISTA]
	@PI_ID_FUTBOLISTA INTEGER,
	@PV_NOMBRE VARCHAR (100),
	@PV_APELLIDOS VARCHAR (100),
	@PV_EDAD VARCHAR (100),
	@PV_DIRECCION VARCHAR (100),
	@PV_MAIL VARCHAR (100),
	@PD_FECHA_NACIMIENTO DATETIME
	
AS
BEGIN
	
	UPDATE FUTBOLISTA
	SET NOMBRE =@PV_NOMBRE,
	APELLIDOS =@PV_APELLIDOS,
	EDAD =@PV_EDAD,
	DIRECCION=@PV_DIRECCION,
	MAIL=@PV_MAIL,
	FECHA_NACIMIENTO=@PD_FECHA_NACIMIENTO 
	WHERE ID_FUTBOLISTA=@PI_ID_FUTBOLISTA

END
